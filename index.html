ページトップへ戻るボタンサンプルだよ。

		
			    $(function () {
				var $nav = $("#sidemenu");
				$(".JS-navHoverBox").on({
				    "mouseenter": function () {
					$nav.stop().animate({"left": 0}, 150);
				    },
				    "mouseleave": function () {
					$nav.stop().animate({"left": -280}, 150);
				    }
				});
			    });
